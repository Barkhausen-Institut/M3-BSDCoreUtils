add_executable(users users.c)

target_link_libraries(users PRIVATE compat)

if(${CMAKE_SYSTEM_NAME} STREQUAL "Linux")
    set(CMAKE_C_FLAGS "${CMAKE_C_FLAGS} -D_GNU_SOURCE")
endif()

file(COPY ${CMAKE_CURRENT_SOURCE_DIR}/users.1 DESTINATION ${CMAKE_CURRENT_BINARY_DIR})

INSTALL(TARGETS users DESTINATION ${CMAKE_INSTALL_BINDIR})

INSTALL(FILES ${CMAKE_CURRENT_BINARY_DIR}/users.1 DESTINATION ${CMAKE_INSTALL_PREFIX}/man/man1)
